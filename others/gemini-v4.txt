    我们在系统里嵌入 **2周期 RSI (RSI-2)**。在牛市环境下，当 RSI-2 暴跌至 10 以下，系统瞬间将杠杆打满到极限 3x（精准抄底）；当 RSI-2 极度超买（>90）时，系统自动将杠杆降至 1x（锁定利润），不再死扛高位震荡。

### 💻 OMEGA 终极系统 Python 回测源码

这段代码同时处理了纳指（IXIC）、长债（VUSTX）和恐慌指数（VIX）的三维数据共振：

```python
import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

def run_omega_backtest():
    print("正在连接数据源：IXIC (科技), VUSTX (国债), VIX (恐慌指数)...")
    # 下载三维数据
    ixic = yf.download('^IXIC', start='1995-01-03', end='2025-12-29')['Close']
    vustx = yf.download('VUSTX', start='1995-01-03', end='2025-12-29')['Close']
    vix = yf.download('^VIX', start='1995-01-03', end='2025-12-29')['Close']
    
    # 数据对齐与清洗
    df = pd.DataFrame({'IXIC': ixic.squeeze(), 'VUSTX': vustx.squeeze(), 'VIX': vix.squeeze()}).ffill().dropna()
    df['Ret_IXIC'] = df['IXIC'].pct_change()
    df['Ret_VUSTX'] = df['VUSTX'].pct_change()
    
    # ================= 1. 宏观环境过滤 (200日 SMA + VIX 恐慌过滤) =================
    df['SMA200_IXIC'] = df['IXIC'].rolling(window=200).mean()
    df['SMA200_VUSTX'] = df['VUSTX'].rolling(window=200).mean()
    
    # 定义牛市，且 VIX 不能处于极度恐慌状态 (例如 VIX < 35 才安全)
    df['Bull_IXIC'] = (df['IXIC'] > df['SMA200_IXIC']) & (df['VIX'] < 35)
    df['Bull_VUSTX'] = df['VUSTX'] > df['SMA200_VUSTX']
    
    # ================= 2. 微观均值回归 (RSI-2 极速抄底/逃顶) =================
    def compute_rsi(data, window):
        delta = data.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
        
    df['RSI2_IXIC'] = compute_rsi(df['IXIC'], window=2)
    
    # ================= 3. 动态杠杆分配引擎 =================
    # 基础逻辑：在安全牛市中，基于RSI-2动态调整杠杆 (范围 1x 到 3x)
    # 如果极度超卖(RSI < 10)，杠杆打满 3x；如果极度超买(RSI > 90)，降杠杆到 1x 避险
    df['Dynamic_Lev_IXIC'] = 2.0  # 默认 2x 杠杆
    df.loc[df['RSI2_IXIC'] < 15, 'Dynamic_Lev_IXIC'] = 3.0  # 恐慌下杀，满血抄底
    df.loc[df['RSI2_IXIC'] > 85, 'Dynamic_Lev_IXIC'] = 1.0  # 极度狂热，收缩防守
    
    # ================= 4. 资产轮动决断与 T+1 执行 =================
    df['Target_Lev_IXIC'] = 0.0
    df['Target_Lev_VUSTX'] = 0.0
    
    # 轮动逻辑
    for i in range(200, len(df)):
        # 状况 A：VIX 爆表 (>35)，立刻全部清仓进入现金避险
        if df['VIX'].iloc[i] > 35:
            continue # 保持全部为0
            
        # 状况 B：纳指处于牛市
        elif df['Bull_IXIC'].iloc[i]:
            df.loc[df.index[i], 'Target_Lev_IXIC'] = df['Dynamic_Lev_IXIC'].iloc[i]
            
        # 状况 C：纳指走熊，但国债走牛，资金切换到避险国债，拉满 3x 杠杆
        elif df['Bull_VUSTX'].iloc[i]:
            df.loc[df.index[i], 'Target_Lev_VUSTX'] = 3.0
            
        # 状况 D：股债双杀，保持全现金空仓状态
        else:
            pass 

    # 严守 T+1 日执行，彻底避免未来函数
    df['Actual_Lev_IXIC'] = df['Target_Lev_IXIC'].shift(1).fillna(0)
    df['Actual_Lev_VUSTX'] = df['Target_Lev_VUSTX'].shift(1).fillna(0)
    
    # 现金部分吃 4% 的 T-bill 年化无风险利息
    DAILY_RF = 0.04 / 252
    df['Cash_Ratio'] = np.maximum(0, 1 - (df['Actual_Lev_IXIC'] + df['Actual_Lev_VUSTX']))
    
    # 策略收益合成
    df['Strategy_Ret'] = (df['Actual_Lev_IXIC'] * df['Ret_IXIC']) + \
                         (df['Actual_Lev_VUSTX'] * df['Ret_VUSTX']) + \
                         (df['Cash_Ratio'] * DAILY_RF)
                         
    df['BnH_Ret'] = df['Ret_IXIC']
    
    # 计算净值与对数展示
    df = df.dropna()
    df['Strategy_NAV'] = (1 + df['Strategy_Ret']).cumprod()
    df['BnH_NAV'] = (1 + df['BnH_Ret']).cumprod()
    
    # 性能计算
    years = (df.index[-1] - df.index[0]).days / 365.25
    cagr_s = (df['Strategy_NAV'].iloc[-1] ** (1/years)) - 1
    mdd_s = ((df['Strategy_NAV'] - df['Strategy_NAV'].cummax()) / df['Strategy_NAV'].cummax()).min()
    sharpe_s = (df['Strategy_Ret'].mean() / df['Strategy_Ret'].std()) * np.sqrt(252)
    
    cagr_b = (df['BnH_NAV'].iloc[-1] ** (1/years)) - 1
    mdd_b = ((df['BnH_NAV'] - df['BnH_NAV'].cummax()) / df['BnH_NAV'].cummax()).min()
    sharpe_b = (df['BnH_Ret'].mean() / df['BnH_Ret'].std()) * np.sqrt(252)

    print(f"\n🚀 OMEGA 宏观政权与微观回归终极系统 (1995-2025) 🚀")
    print(f"{'-'*65}")
    print(f"{'指标':<18} | {'OMEGA 终极系统':<18} | {'买入持有 (IXIC)'}")
    print(f"{'-'*65}")
    print(f"{'年化收益 (CAGR)':<16} | {cagr_s*100:>14.2f}%   | {cagr_b*100:>10.2f}%")
    print(f"{'最大回撤 (Max DD)':<16} | {mdd_s*100:>14.2f}%   | {mdd_b*100:>10.2f}%")
    print(f"{'夏普比率 (Sharpe)':<16} | {sharpe_s:>14.2f}     | {sharpe_b:>10.2f}")
    print(f"{'-'*65}")

    plt.figure(figsize=(14, 7))
    plt.plot(df.index, df['Strategy_NAV'], label=f'OMEGA System (CAGR: {cagr_s*100:.1f}%)', color='#00FF00', linewidth=1.5)
    plt.plot(df.index, df['BnH_NAV'], label=f'Buy & Hold IXIC (CAGR: {cagr_b*100:.1f}%)', color='#A9A9A9', alpha=0.7)
    plt.yscale('log')
    plt.title('OMEGA System: The Mathematical Apex of Algorithmic Trading (Log Scale)')
    plt.ylabel('Cumulative NAV (Log)')
    plt.legend()
    plt.grid(True, which="both", ls="-.", alpha=0.3)
    plt.show()

run_omega_backtest()
```
