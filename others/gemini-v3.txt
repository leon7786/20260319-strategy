### 🌌 TRKE 系统的降维打击逻辑

我们引入一个完美的防御反击资产：**美国长期国债（选用 VUSTX，先锋长期国债基金，完美覆盖 1995-2025 年）**。历史上，当科技股崩盘（2000年科网泡沫、2008年次贷危机、2020年疫情）时，避险资金会疯狂涌入长期国债，导致国债暴涨。

这套系统将不再是“躲避暴跌”，而是**“在科技股暴跌时，通过做多国债继续暴赚”**。

1.  **绝对趋势过滤（双重 200 日均线）：** 分别计算纳指和国债的 200 日均线。只有价格在均线之上，该资产才拥有“上桌”的资格。
2.  **相对动能锦标赛（90 日角斗场）：** 如果纳指和国债都在牛市（都在 200 日均线之上），系统会计算它们过去 90 天的收益率。**强者通吃**，资金只买入动能更强的那一个。
3.  **2022 级通胀地堡（现金避风港）：** 当股债双杀（例如 2022 年美联储极端加息，纳指和国债双双跌破 200 日均线）时，系统立刻空仓，全额转入无风险的短期国库券（T-bills）赚取利息。
4.  **波动率目标缩放（最高 3x）：** 选定赢家后，根据其近 20 日的实际波动率，套用公式 $Leverage = \frac{Target\_Vol}{\sigma_{20}}$，动态分配 0x 到 3x 的杠杆。

### 💻 TRKE 双资产轮动系统 Python 源码

这套代码整合了双资产下载、数据对齐、锦标赛轮动和 T+1 严格执行，彻底根绝未来函数：

```python
import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

def run_trke_backtest():
    print("正在下载 IXIC (科技攻) 与 VUSTX (国债防) 的日频数据...")
    # 分别下载并提取收盘价
    ixic = yf.download('^IXIC', start='1995-01-03', end='2025-12-29')['Close']
    vustx = yf.download('VUSTX', start='1995-01-03', end='2025-12-29')['Close']
    
    # 将两个序列合并为一个 DataFrame，去掉列名的 MultiIndex 干扰，并前向填充处理交易日差异
    df = pd.DataFrame({'IXIC': ixic.squeeze(), 'VUSTX': vustx.squeeze()}).ffill().dropna()
    
    # 每日收益率
    df['Ret_IXIC'] = df['IXIC'].pct_change()
    df['Ret_VUSTX'] = df['VUSTX'].pct_change()
    
    # ================= 1. 绝对趋势过滤 (200日 SMA) =================
    df['SMA200_IXIC'] = df['IXIC'].rolling(window=200).mean()
    df['SMA200_VUSTX'] = df['VUSTX'].rolling(window=200).mean()
    
    df['Bull_IXIC'] = df['IXIC'] > df['SMA200_IXIC']
    df['Bull_VUSTX'] = df['VUSTX'] > df['SMA200_VUSTX']
    
    # ================= 2. 相对动能锦标赛 (90日收益率) =================
    df['Mom90_IXIC'] = df['IXIC'].pct_change(90)
    df['Mom90_VUSTX'] = df['VUSTX'].pct_change(90)
    
    # 逻辑判断：到底选谁？(0=现金, 1=纳指, 2=国债)
    # 默认持有现金
    df['Target_Asset'] = 0 
    
    # 只有一方在牛市，直接选它
    df.loc[df['Bull_IXIC'] & ~df['Bull_VUSTX'], 'Target_Asset'] = 1
    df.loc[~df['Bull_IXIC'] & df['Bull_VUSTX'], 'Target_Asset'] = 2
    
    # 都在牛市，选90天动能更强的
    both_bull = df['Bull_IXIC'] & df['Bull_VUSTX']
    df.loc[both_bull & (df['Mom90_IXIC'] >= df['Mom90_VUSTX']), 'Target_Asset'] = 1
    df.loc[both_bull & (df['Mom90_IXIC'] < df['Mom90_VUSTX']), 'Target_Asset'] = 2
    
    # ================= 3. 波动率目标自适应杠杆 =================
    TARGET_VOL = 0.35  # 目标年化波动率 35%
    MAX_LEV = 3.0      # 最高 3x 杠杆
    
    df['Vol20_IXIC'] = df['Ret_IXIC'].rolling(window=20).std() * np.sqrt(252)
    df['Vol20_VUSTX'] = df['Ret_VUSTX'].rolling(window=20).std() * np.sqrt(252)
    
    # 避免除以0
    df['Vol20_IXIC'] = df['Vol20_IXIC'].replace(0, np.nan).fillna(method='bfill')
    df['Vol20_VUSTX'] = df['Vol20_VUSTX'].replace(0, np.nan).fillna(method='bfill')
    
    df['Lev_IXIC'] = (TARGET_VOL / df['Vol20_IXIC']).clip(0, MAX_LEV)
    df['Lev_VUSTX'] = (TARGET_VOL / df['Vol20_VUSTX']).clip(0, MAX_LEV)
    
    # ================= 4. T+1 执行与收益计算 =================
    # 根据目标资产，获取 T 日的理论杠杆
    df['Target_Lev_IXIC_T'] = np.where(df['Target_Asset'] == 1, df['Lev_IXIC'], 0)
    df['Target_Lev_VUSTX_T'] = np.where(df['Target_Asset'] == 2, df['Lev_VUSTX'], 0)
    
    # 严格延后一天执行 (防未来函数)
    df['Actual_Lev_IXIC'] = df['Target_Lev_IXIC_T'].shift(1).fillna(0)
    df['Actual_Lev_VUSTX'] = df['Target_Lev_VUSTX_T'].shift(1).fillna(0)
    
    # 闲置资金去买 T-Bills (按年化 4% 算)
    DAILY_RF = 0.04 / 252
    df['Cash_Ratio'] = np.maximum(0, 1 - (df['Actual_Lev_IXIC'] + df['Actual_Lev_VUSTX']))
    
    # 最终收益 = 纳指部分 + 国债部分 + 现金利息部分
    df['Strategy_Ret'] = (df['Actual_Lev_IXIC'] * df['Ret_IXIC']) + \
                         (df['Actual_Lev_VUSTX'] * df['Ret_VUSTX']) + \
                         (df['Cash_Ratio'] * DAILY_RF)
                         
    df['BnH_Ret'] = df['Ret_IXIC'] # 买入持有纳指作为基准
    
    # 计算净值
    df = df.dropna()
    df['Strategy_NAV'] = (1 + df['Strategy_Ret']).cumprod()
    df['BnH_NAV'] = (1 + df['BnH_Ret']).cumprod()
    
    # ================= 5. 性能评估 =================
    years = (df.index[-1] - df.index[0]).days / 365.25
    
    def get_stats(returns, navs):
        cagr = (navs.iloc[-1] ** (1/years)) - 1
        max_dd = ((navs - navs.cummax()) / navs.cummax()).min()
        sharpe = (returns.mean() / returns.std()) * np.sqrt(252)
        return cagr, max_dd, sharpe

    c_s, m_s, sh_s = get_stats(df['Strategy_Ret'], df['Strategy_NAV'])
    c_b, m_b, sh_b = get_stats(df['BnH_Ret'], df['BnH_NAV'])

    print(f"\n💎💎💎 TRKE 三位一体多资产轮动回测 (1995-2025) 💎💎💎")
    print(f"{'-'*60}")
    print(f"{'指标':<18} | {'TRKE 轮动系统':<15} | {'买入持有 (IXIC)'}")
    print(f"{'-'*60}")
    print(f"{'年化收益 (CAGR)':<16} | {c_s*100:>12.2f}%   | {c_b*100:>10.2f}%")
    print(f"{'最大回撤 (Max DD)':<16} | {m_s*100:>12.2f}%   | {m_b*100:>10.2f}%")
    print(f"{'夏普比率 (Sharpe)':<16} | {sh_s:>12.2f}    | {sh_b:>10.2f}")
    print(f"{'-'*60}")

    plt.figure(figsize=(14, 7))
    plt.plot(df.index, df['Strategy_NAV'], label=f'TRKE System (CAGR: {c_s*100:.1f}%)', color='#8A2BE2', linewidth=1.5)
    plt.plot(df.index, df['BnH_NAV'], label=f'Buy & Hold IXIC (CAGR: {c_b*100:.1f}%)', color='#A9A9A9', alpha=0.7)
    plt.yscale('log')
    plt.title('The Holy Grail of Diversification: TRKE Multi-Asset System (Log Scale)')
    plt.ylabel('Cumulative NAV (Log)')
    plt.legend()
    plt.grid(True, which="both", ls="--", alpha=0.4)
    plt.show()

run_trke_backtest()
```
